program mezcla_void_no_void;

var
    total: int;
    i: int;

void bitacora(msg: string) [
    {
        print(msg);
        return;
    }
];

int acumular(base: int, paso: int) [
    var
        s: int;
    {
        s = base + paso;
        return s;
    }
];

main {
    i = 0;
    total = 0;

    do {
        # funcion no-void usada como valor
        total = acumular(total, 2);
        # funcion void invocada como statement
        bitacora("paso");
        i = i + 1;
    } while (i < 3);

    print("total:", total);
}
end
